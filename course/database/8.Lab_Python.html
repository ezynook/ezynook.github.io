การใช้ฟังก์ชันของ pandas ร่วมกับ SQLAlchemy ทำให้การจัดการข้อมูลและการทำงานกับฐานข้อมูลเป็นเรื่องง่ายและมีประสิทธิภาพ ตัวอย่างด้านล่างจะแสดงวิธีการใช้ฟังก์ชันต่าง ๆ ของ pandas ร่วมกับการทำงานกับฐานข้อมูล SQL โดยใช้ SQLAlchemy

การเชื่อมต่อกับฐานข้อมูลและดึงข้อมูล

import pandas as pd
from sqlalchemy import create_engine

# สร้างการเชื่อมต่อฐานข้อมูล SQLite
engine = create_engine(
    'sqlite:///./File/demo.db'
)
# ใช้ pandas ดึงข้อมูลจากฐานข้อมูล
df_customers = pd.read_sql_table(
    'customers', 
    con=engine
)
df_orders = pd.read_sql_table(
    'orders', 
    con=engine
)

การใช้ฟังก์ชัน df.head()

df_customers.head()

การใช้ฟังก์ชัน df.tail()

df_orders.tail()

การใช้ฟังก์ชัน df.info()

df_customers.info()

การใช้ฟังก์ชัน df.query() (กรองข้อมูลด้วยเงื่อนไขที่ระบุ)

filtered_df = df_orders.query(
    'amount > 100.00'
)
filtered_df

การใช้ฟังก์ชัน df.loc[] และ df.iloc[] (เข้าถึงข้อมูลตามตำแหน่งหรือตามเงื่อนไข)

# ใช้ loc[] เพื่อเลือกแถวที่มี customer_id = 1
selected_row = df_customers.loc[
    df_customers['customer_id'] == 1
]
print(selected_row)

# ใช้ iloc[] เพื่อเลือกแถวที่ตำแหน่งที่ระบุ
selected_row_by_index = df_customers.iloc[0]
print(selected_row_by_index)

การใช้ฟังก์ชัน df.groupby() (จัดกลุ่มข้อมูลและคำนวณสถิติ)

grouped_df = df_orders.groupby('customer_id').sum()
print(grouped_df)

การใช้ฟังก์ชัน df.merge() (รวมข้อมูลจากหลาย DataFrame *JOIN)

merged_df = pd.merge(
    df_customers, 
    df_orders, 
    left_on='customer_id', 
    right_on='customer_id'
)
print(merged_df)

การใช้ฟังก์ชัน df.drop() (ลบคอลัมน์หรือแถว)

df_dropped = df_customers.drop(
    columns=['last_name']
)
print(df_dropped)

การใช้ฟังก์ชัน df.rename() (เปลี่ยนชื่อคอลัมน์)

df_renamed = df_customers.rename(
    columns = {
        'first_name': 'fname'
    }
)
print(df_renamed)

การใช้ฟังก์ชัน df.fillna() (แทนที่ค่า NaN ด้วยค่าที่ระบุ)

df_filled = df_orders.fillna(0)
print(df_filled)

การใช้ฟังก์ชัน df.to_sql() (เขียน DataFrame ไปยังฐานข้อมูล SQL)

df_customers.to_sql(
    'customers_test', 
    con=engine, 
    if_exists='replace', 
    index=False
)

การใช้ฟังก์ชัน df.to_csv()

df_orders.to_csv(
    'orders.csv', 
    index=False
)
