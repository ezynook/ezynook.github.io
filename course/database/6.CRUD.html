CRUD Statement

ในการทำงานกับฐานข้อมูล SQLite คุณสามารถใช้คำสั่ง SQL สำหรับการทำงาน CRUD (Create, Read, Update, Delete) ดังนี้:

1. Create (การสร้างข้อมูล)

INSERT INTO
    my_table (
        name, age, email
    ) VALUES (
        'John Doe', 30, 'john@example.com'
    );

2. Read (การอ่านข้อมูล)

SELECT * FROM customers
--ดึงข้อมูลเฉพาะคอลัมน์:
SELECT first_name, last_name, email FROM my_table;

WHERE การใช้คำสั่ง SELECT ร่วมกับ WHERE ใน SQL เป็นวิธีการกรองข้อมูลจากตารางตามเงื่อนไขต่าง ๆ มีหลายรูปแบบที่สามารถใช้ได้ ดังนี้:

  1. การใช้ตัวดำเนินการเท่ากับ (=)
SELECT * FROM customers WHERE customer_id = 1;
  1. การใช้ตัวดำเนินการไม่เท่ากับ (!= หรือ <>)
SELECT * FROM customers WHERE customer_id != 1;
  1. การใช้ตัวดำเนินการมากกว่า (>) และน้อยกว่า (<)
SELECT * FROM orders WHERE amount > 100.00;
SELECT * FROM orders WHERE amount < 100.00;
  1. การใช้ตัวดำเนินการมากกว่าหรือเท่ากับ (>=) และน้อยกว่าหรือเท่ากับ (<=)
SELECT * FROM orders WHERE amount >= 100.00;
SELECT * FROM orders WHERE amount <= 100.00;
  1. การใช้คำสั่ง BETWEEN
SELECT * FROM orders WHERE amount BETWEEN 100.00 AND 200.00;
  1. การใช้คำสั่ง IN
SELECT * FROM customers WHERE customer_id IN (1, 2, 3);
  1. การใช้คำสั่ง NOT IN
SELECT * FROM customers WHERE customer_id NOT IN (1, 2, 3);
  1. การใช้คำสั่ง LIKE (เลือกข้อมูลลูกค้าที่ชื่อขึ้นต้นด้วยตัวอักษร “J”)
SELECT * FROM customers WHERE first_name LIKE 'J%';
  1. การใช้คำสั่ง NOT LIKE (เลือกข้อมูลลูกค้าที่ชื่อไม่ขึ้นต้นด้วยตัวอักษร “J”)
SELECT * FROM customers WHERE first_name NOT LIKE 'J%';
  1. การใช้คำสั่ง IS NULL และ IS NOT NULL (เลือกข้อมูลคำสั่งซื้อที่ amount เป็น NULL)
SELECT * FROM orders WHERE amount IS NULL;
SELECT * FROM orders WHERE amount IS NOT NULL;
  1. การใช้คำสั่ง EXISTS (เลือกข้อมูลลูกค้าที่มีคำสั่งซื้ออยู่ในตาราง orders)
SELECT * FROM customers WHERE EXISTS (SELECT 1 FROM orders WHERE customers.customer_id = orders.customer_id);
  1. การใช้คำสั่ง NOT EXISTS
SELECT * FROM customers WHERE NOT EXISTS (SELECT 1 FROM orders WHERE customers.customer_id = orders.customer_id);
  1. การใช้คำสั่ง AND (เลือกข้อมูลคำสั่งซื้อที่มีจำนวน (amount) มากกว่า 100.00 และถูกสั่งซื้อในวันที่ 10 มกราคม 2024)
SELECT * FROM orders WHERE amount > 100.00 AND order_date = '2024-01-10';
  1. การใช้คำสั่ง OR (เลือกข้อมูลคำสั่งซื้อที่มีจำนวน (amount) มากกว่า 100.00 หรือถูกสั่งซื้อในวันที่ 10 มกราคม 2024)
SELECT * FROM orders WHERE amount > 100.00 OR order_date = '2024-01-10';
  1. การใช้คำสั่ง CASE ใน WHERE
SELECT * FROM orders WHERE 
CASE 
    WHEN order_date >= '2024-01-01' THEN amount > 100.00
    ELSE amount > 200.00
END;

INNER JOIN INNER JOIN จะคืนค่าบรรทัดที่มีการจับคู่ข้อมูลในทั้งสองตารางเท่านั้น

SELECT c.customer_id, c.first_name, o.order_id, o.amount
FROM customers c
INNER JOIN orders o ON c.customer_id = o.customer_id;

ผลลัพธ์: คืนค่าข้อมูลเฉพาะลูกค้าที่มีคำสั่งซื้อ (เฉพาะบรรทัดที่มีการจับคู่ในทั้งสองตาราง)

LEFT JOIN LEFT JOIN จะคืนค่าบรรทัดทั้งหมดจากตารางทางซ้าย (ตาราง customers) และค่าที่ตรงกันจากตารางทางขวา (ตาราง orders) ถ้าไม่มีการจับคู่จะคืนค่า NULL สำหรับคอลัมน์ของตารางขวา

SELECT c.customer_id, c.first_name, o.order_id, o.amount
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id;

ผลลัพธ์: คืนค่าข้อมูลทั้งหมดของลูกค้า แม้ว่าจะไม่มีคำสั่งซื้อก็ตาม (NULL จะถูกคืนค่าในคอลัมน์ของตาราง orders ที่ไม่มีการจับคู่)

RIGHT JOIN RIGHT JOIN จะคืนค่าบรรทัดทั้งหมดจากตารางทางขวา (ตาราง orders) และค่าที่ตรงกันจากตารางทางซ้าย (ตาราง customers) ถ้าไม่มีการจับคู่จะคืนค่า NULL สำหรับคอลัมน์ของตารางซ้าย

SELECT c.customer_id, c.first_name, o.order_id, o.amount
FROM customers c
RIGHT JOIN orders o ON c.customer_id = o.customer_id;

ผลลัพธ์: คืนค่าข้อมูลทั้งหมดของคำสั่งซื้อ แม้ว่าจะไม่มีลูกค้าก็ตาม (NULL จะถูกคืนค่าในคอลัมน์ของตาราง customers ที่ไม่มีการจับคู่)

3. Update (การแก้ไขข้อมูล)

UPDATE my_table
SET age = 31
WHERE name = 'John Doe';
---การอัปเดตหลายคอลัมน์พร้อมกัน:
UPDATE my_table
SET age = 32, email = 'john.new@example.com'
WHERE name = 'John Doe';

4. Delete (การลบข้อมูล)

DELETE FROM my_table
WHERE name = 'John Doe';
---การลบข้อมูลทั้งหมดในตาราง
DELETE FROM my_table;

5. DROP TABLE

DROP TABLE table_name;
